[documentos]    def dividir(self, a:int, b:int)->int:
        """Realiza la división entera de a entre b, sólo si la división es exacta.
        Si no es exacta devuelve ValueError.
        Si el divisor es cero(0) devolvemos Zero

        :param a: dividendo
        :type a: int
        :param b: divisor
        :type b: int
        :return: cociente
        :rtype: int

        """

        if b == 0:
            return ZeroDivisionError

        if a % b != 0:
            return ValueError
        else:
            return a // b  # En Python 2.x esto funciona sin problemas, pero en Python 3, no. ¿Sabes arreglarlo?
